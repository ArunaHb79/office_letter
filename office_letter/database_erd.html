Entity Relationship Diagram (ERD)

erDiagram
    Department ||--o{ Employee : "has"
    Department ||--o{ Letter : "handles"
    Employee ||--|| Users : "has account"
    Employee ||--o{ Letter : "assigned to"
    Employee ||--o{ password_resets : "can request"
    Users ||--o{ ActivityLog : "performs"
    Letter ||--o{ ActivityLog : "tracked by"
    LetterMethod ||--o{ Letter : "used by"
    LetterStatus ||--o{ Letter : "applies to"

    Department {
        int id PK
        varchar name
        varchar abbreviation
    }

    Employee {
        int id PK
        varchar name
        varchar email UK
        varchar password
        int department_id FK
        varchar role
    }

    Users {
        int id PK
        varchar username
        varchar password
        int employee_id FK
        varchar role
        varchar email
    }

    Letter {
        int id PK
        text subject
        text content
        varchar sender
        varchar receiver
        int method_id FK
        date date_received
        date date_sent
        int department_id FK
        int employee_id FK
        varchar letter_number UK
        int status_id FK
    }

    LetterMethod {
        int id PK
        varchar method_name
    }

    LetterStatus {
        int id PK
        varchar status_name
    }

    ActivityLog {
        int id PK
        int user_id FK
        int letter_id FK "nullable"
        varchar action
        json old_values
        json new_values
        varchar ip_address
        text user_agent
        timestamp timestamp
    }

    password_resets {
        int id PK
        int employee_id FK
        varchar token UK
        datetime expires_at
        datetime created_at
    }
